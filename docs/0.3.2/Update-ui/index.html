    Update UI

    Receive status updates

    AutomixIQ reports its current state and any playback event via AutomixSDKStatusListener. This listener reports the status via its statusChanged method every time the status changed, and in the function it specifies the current status and the variables that were altered since the last time an update was issued. Use the status variables to update the content of your player UI. Here is an example implementation:

    public class MyPlayer: AutomixSDKStatusListener {
    // ...

    var latestStatus: AutomixSDKStatus? = null
    override fun statusChanged(variables:List<AutomixSDKStatusVariable>, status: AutomixSDKStatus) {
        this.latestStatus = status
        for (property in variables) {    
            when (property) {
                AutomixSDKStatusVariable.PLAYING -> {
                    // Change the look and feel of yor Play/Pause button here
                    val playing = status.playing
                    // ...
                }
                AutomixSDKStatusVariable.MIXING -> {}
                AutomixSDKStatusVariable.PLAYING_TRACK_IDENTIFIER -> {}
                AutomixSDKStatusVariable.PLAYING_TRACK_INDEX -> {}
                AutomixSDKStatusVariable.PLAYING_TRACK_CURRENT_TIME -> {
                    // Change your track progress UI here 
                    val ellapsedTime = status.playingTrackCurrentTime
                    val totalTime = status.playingTrackTotalTime
                    // ...
                }
                AutomixSDKStatusVariable.PLAYING_TRACK_STREAMING_PROGRESS -> {}
                AutomixSDKStatusVariable.PLAYING_TRACK_TOTAL_TIME -> {
                    // Change your track progress UI here 
                    val ellapsedTime = status.playingTrackCurrentTime
                    val totalTime = status.playingTrackTotalTime
                    // ...
                }
                AutomixSDKStatusVariable.CAN_SKIP_TO_PREVIOUS -> {
                    // Change the look and feel of yor Previous button here
                    val canSkipToPrevious = status.canSkipToPrevious
                    // ...
                }
                AutomixSDKStatusVariable.CAN_SKIP_TO_NEXT -> {
                    // Change the look and feel of yor Next button here
                    val canSkipToNext = status.canSkipToNext
                    // ...
                }
                AutomixSDKStatusVariable.CAN_PLAY_PAUSE -> {
                    // Change the look and feel of yor Play/Pause button here
                    val canPlayPause = status.canPlayPause
                    // ...
                }
                AutomixSDKStatusVariable.CAN_SEEK -> {}
                AutomixSDKStatusVariable.CAN_REORDER_TRACKS -> {}
                AutomixSDKStatusVariable.REPLAY_ON -> {
                    // Change the look and feel of yor Replay button here
                    val replayOn = status.replayOn
                    // ...
                }
                AutomixSDKStatusVariable.MUTED -> {}
                AutomixSDKStatusVariable.MASTER_TEMPO_NORMALISED -> {}
                AutomixSDKStatusVariable.NORMALISED_VOLUME -> {}
                AutomixSDKStatusVariable.REQUESTED_LOOP_TRACK_IDENTIFIER -> {}
                AutomixSDKStatusVariable.PLAYING_TRACK_INDEX_INTERNAL -> {}
                AutomixSDKStatusVariable.PLAYING_TRACK_CURRENT_TIME_INTERNAL -> {}
                AutomixSDKStatusVariable.CAN_BACK -> {}
                AutomixSDKStatusVariable.READY_TO_PLAY -> {}
                AutomixSDKStatusVariable.CAN_MIX_NOW -> {}
                AutomixSDKStatusVariable.PLAYLIST -> {}
                AutomixSDKStatusVariable.PLAY_SPEED -> {}
                AutomixSDKStatusVariable.PREFERRED_TRANSITION_DURATION_SECONDS -> {}
                AutomixSDKStatusVariable.REQUESTED_PAUSE_TRACK_IDENTIFIER -> {}
                AutomixSDKStatusVariable.VINYL_START_STOP -> {}
                AutomixSDKStatusVariable.NEXT_TRANSITION_DURATION -> {}
                AutomixSDKStatusVariable.TRANSITION_START_TIME_SECONDS -> {}
                AutomixSDKStatusVariable.FAILED_TRACK_LOAD_IDENTIFIERS -> {}
                AutomixSDKStatusVariable.CAN_ON_OFF -> {}
                AutomixSDKStatusVariable.PLAYING_TRACK_ORIGINAL_SESSION_ID -> {}
                AutomixSDKStatusVariable.MIX_ID -> {}
            }
        }
    }


    // ...
}

    Receive playback events

    To listen to playback events, implement the playbackEvent method.

    public class MyPlayer: AutomixSDKStatusListener {
    // ...

    override fun playbackEvent(type: AutomixSDKPlaybackEventType, identifier: String?, originalSessionId: String?, status: AutomixSDKStatus?) {
        when (type) {
            AutomixSDKPlaybackEventType.trackPlayStart -> {}
            AutomixSDKPlaybackEventType.trackPlayPlayed5Secs -> {}
            AutomixSDKPlaybackEventType.trackPlayHeartbeat30Secs -> {}
            AutomixSDKPlaybackEventType.trackPlaySkipped -> {}
            AutomixSDKPlaybackEventType.trackPlayFinished -> {}
            AutomixSDKPlaybackEventType.mixPlaybackEnded -> {}
            AutomixSDKPlaybackEventType.trackPlayStartMixingIntoNext -> {}
        }
    }

    // ...
}

    The type of playback event is modelled by the enum AutomixSDKPlaybackEventType.
